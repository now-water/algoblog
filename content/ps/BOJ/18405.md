import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

class Virus{
    int x, y, sort;

    public Virus(int x, int y, int sort) {
        this.x = x;
        this.y = y;
        this.sort = sort;
    }
}
public class Q18405 {
    static int n, k, s, x, y, arr[][];
    static int dx[] = {-1, 0, 1, 0}, dy[] = {0, 1, 0, -1};
    static Queue<Virus> queues[];
    static boolean[][] visited;

    public static void main(String[] args) {
        init();
        solution();
    }

    static void solution() {
        int time = 1;
        /* ë‚®ì€ ë²ˆí˜¸ìˆœìœ¼ë¡œ ì „ì—¼ */
        while(time <= s) {
            for (int i = 1; i <= k; i++)
                contaminate(queues[i]);
            time += 1;
        }
        System.out.println(arr[x][y]);
    }

    static void contaminate(Queue<Virus> q) {
        int size = q.size();
        for (int si = 0; si < size; si++) {
            Virus it = (Virus) q.poll();
            for (int d = 0; d < 4; d++) {
                int nx = it.x + dx[d], ny = it.y + dy[d];
                if (!isBorder(nx, ny) || visited[nx][ny])
                    continue;
                arr[nx][ny] = it.sort;
                visited[nx][ny] = true;
                q.add(new Virus(nx, ny, it.sort));
            }
        }
    }

    static boolean isBorder(int x, int y){
        return (x >= 0 && x < n && y >= 0 && y < n);
    }
    static void init() {
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        k = sc.nextInt();
        arr = new int[n][n];
        visited = new boolean[n][n];
        queues = new Queue[k + 1];
        for (int i = 0; i <= k; i++)
            queues[i] = new LinkedList<Virus>();

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int num = arr[i][j] = sc.nextInt();
                if(num != 0) {
                    queues[num].add(new Virus(i, j, num));
                    visited[i][j] = true;
                }
            }
        }
        s = sc.nextInt();
        x = sc.nextInt() - 1;
        y = sc.nextInt() - 1;
    }
}
